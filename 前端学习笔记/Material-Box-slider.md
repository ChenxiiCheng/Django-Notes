       <li>
         <img src="img/img3.jpg" />
         <!-- CENTER ALIGN CAPTION -->
         <div class="caption center-align">
           <h3 class="black-text">Center Caption</h3>
           <h5 class="black-text">
             Lorem ipsum dolor sit amet.
           </h5>
